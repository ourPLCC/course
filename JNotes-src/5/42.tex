define c1 =
  class
    method m1 = proc() send self m2()
    method m2 = proc() 13
  end
define c2 =
  class extends c1
    method m1 = proc() 22
    method m2 = proc() 23
    method m3 = proc() send super m1()
  end
define c3 =
  class extends c2
    method m1 = proc() 32
    method m2 = proc() 33
  end
define o3 = new c3

send o3 m3()  % returns 33
